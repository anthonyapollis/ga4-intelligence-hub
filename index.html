-- UNNEST pattern for ecommerce items
SELECT event_date, user_pseudo_id,
  (SELECT value.string_value FROM UNNEST(event_params) WHERE key='transaction_id') AS transaction_id,
  items.item_id, items.item_name, items.quantity, items.price
FROM `project.analytics_XXX.events_*`,
  UNNEST(items) AS items
WHERE _TABLE_SUFFIX BETWEEN '20230101' AND '20231031'
  AND event_name = 'purchase';

4 dbt Data Quality Tests Applied

not_null
Flags: event_date, user_pseudo_id, event_name, revenue (purchase events only)
unique
Ensures: transaction_id unique in fact_orders; ga_session_id unique in fact_sessions
accepted_values
Validates: event_name in [page_view, purchase, add_to_cart...]; currency in [USD, EUR, GBP]
relationships
Checks: every transaction_id in items exists in fact_orders; every session_id in events exists in sessions
# models/staging/schema.yml
models:
  - name: stg_ga4_events
    columns:
      - name: event_date
        tests: [not_null]
      - name: event_name
        tests:
          - not_null
          - accepted_values:
            values: [page_view, purchase, add_to_cart, begin_checkout, view_item, ...]
  - name: fact_orders
    columns:
      - name: transaction_id
        tests: [not_null, unique]
      - name: revenue
        tests: [not_null, dbt_expectations.expect_column_values_to_be_between(0, 10000)]
